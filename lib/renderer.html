      start: (callback) ->
    if @active then @stop()
    @active = true
    @loop callback
    return this

  stop: ->
    @active = false
    if @frameId then cancelAnimationFrame @frameId
    @frameId = null
    return this

  loop: (callback = ->) ->
    if @active
      @frameId = requestAnimationFrame (time) =>
        @thisFrameTime = time
        @previousFrameTime ||= time
        elapsed = @thisFrameTime - @previousFrameTime
        if @active
          callback time, elapsed
          @loop callback
        else
          cancelAnimationFrame @frameId
    return this
